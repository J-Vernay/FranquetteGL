57/// Bitset representing the interest of user in the different message types.
58enum FEGL_DebugTypeMask {
59 fegl_DEBUG_CRITICAL_BIT = 1, ///< Internal EGL failures (BAD_ALLOC, CONTEXT_LOST)
60 fegl_DEBUG_ERROR_BIT = 2, ///< Input and bad match errors (BAD_PARAMETER...)
61 fegl_DEBUG_WARN_BIT = 4, ///< Warnings, deprecated or inefficient operation
62 fegl_DEBUG_INFO_BIT = 8, ///< Verbose, object creation or change in state
63};
64
65/// Conversion from EGL enumeration values to FEGL_debugMask. Returns 0 on unknown value.
66///
67/// Possible values:
68/// - EGL_DEBUG_MSG_CRITICAL_KHR
69/// - EGL_DEBUG_MSG_ERROR_KHR
70/// - EGL_DEBUG_MSG_WARN_KHR
71/// - EGL_DEBUG_MSG_INFO_KHR
72inline EGLint fegl_GetDebugTypeMask(EGLenum messageType)
73{
74 switch (messageType) {
75 case EGL_DEBUG_MSG_CRITICAL_KHR: return fegl_DEBUG_CRITICAL_BIT;
76 case EGL_DEBUG_MSG_ERROR_KHR: return fegl_DEBUG_ERROR_BIT;
77 case EGL_DEBUG_MSG_WARN_KHR: return fegl_DEBUG_WARN_BIT;
78 case EGL_DEBUG_MSG_INFO_KHR: return fegl_DEBUG_INFO_BIT;
79 }
80 return 0;
81}
