16/// Returns an EGLDisplay from a platform-specific display.
17/// When calling it multiple times with same arguments, the same EGLDisplay is returned.
18/// @param platform Which platform is associated to @a native_display .
19/// The set of supported platforms can be retrieved by calling
20/// `eglQueryString(EGL_NO_DISPLAY, EGL_EXTENSIONS)`.
21/// @param native_display Pointer to a platform-specific native object
22/// @param attrib_list Extra arguments required by the platform.
23/// Either NULL (empty) or it must point to an array of EGLAttrib,
24/// which must be terminated by EGL_NONE .
25/// @retval "Valid EGLDisplay" On success.
26/// @retval EGL_NO_DISPLAY On failure.
27///
28/// @FGL
29/// - @a platform must be EGL_NONE.
30/// - @a native_display must be NULL.
31/// - @a attrib_list must be empty.
32/// - All valid calls will return the same EGLDisplay.
33///
34/// @EGL
35/// - 1.5 Complete
36EGLDisplay eglGetPlatformDisplay(EGLenum platform, void* native_display, EGLAttrib const* attrib_list)
37{
38 if (platform != EGL_NONE || native_display != NULL) {
39 fegl_Error(EGL_BAD_PARAMETER, __func__, NULL, NULL);
40 return EGL_NO_DISPLAY;
41 }
42
43 if (attrib_list != NULL && *attrib_list != EGL_NONE) {
44 fegl_Error(EGL_BAD_ATTRIBUTE, __func__, NULL, NULL);
45 return EGL_NO_DISPLAY;
46 }
47
48 fegl_Success();
49 return _fegl_ONLY_DISPLAY;
50}

52/// Returns an EGLDisplay from an implementation-specific ID.
53/// When calling it multiple times with same ID, the same EGLDisplay is returned.
54/// @param display_id Implementation-specific value, which may be EGL_DEFAULT_DISPLAY.
55/// @retval "Valid EGLDisplay" On success.
56/// @retval EGL_NO_DISPLAY On failure.
57///
58/// @FGL
59/// - @a display_id must be EGL_DEFAULT_DISPLAY.
60/// - All valid calls will return the same EGLDisplay.
61EGLDisplay eglGetDisplay(EGLNativeDisplayType display_id)
62{
63 if (display_id != EGL_DEFAULT_DISPLAY) {
64 fegl_Success();
65 return EGL_NO_DISPLAY;
66 }
67 fegl_Success();
68 return _fegl_ONLY_DISPLAY;
69}
