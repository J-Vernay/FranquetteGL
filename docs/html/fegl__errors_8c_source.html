38void fegl_Error(EGLenum error, char const* command, EGLLabelKHR objectLabel, char const* message)
39{
40 _fegl_lastStatus = error;
41 if (_fegl_debugProc == NULL)
42 return; // debug messages disabled
43
44 EGLint messageType = 0;
45 switch (error) {
46 case EGL_SUCCESS: messageType = EGL_DEBUG_MSG_INFO_KHR; break;
47 case EGL_BAD_ALLOC:
48 case EGL_CONTEXT_LOST: messageType = EGL_DEBUG_MSG_CRITICAL_KHR; break;
49 default: messageType = EGL_DEBUG_MSG_ERROR_KHR; break;
50 }
51
52 if ((fegl_GetDebugTypeMask(messageType) & _fegl_debugFlags) == 0)
53 return; // filtered out
54 (*_fegl_debugProc)(error, command, messageType, _fegl_threadLabel, objectLabel, message);
55}

59/// @addtogroup EGL_KHR_debug
60/// @sa - Specification: https://registry.khronos.org/EGL/extensions/KHR/EGL_KHR_debug.txt
61/// @{
62
63/// Controls debug output functionality. Debug messages are passed to @a callback.
64/// Messages can be filtered by types with @a attrib_list .
65/// @param callback User function to be called when a message is outputted.
66/// If NULL, the debug messages are disabled and the type filter is reset.
67/// @param attrib_list Modifies the current type filter.
68/// Expects a list of pairs (KEY,VALUE) terminated by EGL_NONE.
69///
70/// For each (KEY,VALUE) in @a attrib_list , KEY is the message type to filter,
71/// and VALUE is either EGL_TRUE to enable it or EGL_FALSE to disable it.
72/// | Accepted keys for @a attrib_list ||Default|
73/// |----------------------------------||-------|
74/// | EGL_DEBUG_MSG_CRITICAL_KHR | Internal EGL driver failures, i.e. EGL_BAD_ALLOC, EGL_CONTEXT_LOST | EGL_TRUE |
75/// | EGL_DEBUG_MSG_ERROR_KHR | Input and bad match errors, i.e. EGL_BAD_CONTEXT, EGL_BAD_PARAMETER... | EGL_TRUE |
76/// | EGL_DEBUG_MSG_WARN_KHR | Warnings, code is EGL_SUCCESS, but deprecated or inefficient operation | EGL_FALSE |
77/// | EGL_DEBUG_MSG_INFO_KHR | Verbose operations, object creation, destruction, state change... | EGL_FALSE |
78///
79EGLint eglDebugMessageControlKHR(EGLDEBUGPROCKHR callback, EGLAttrib const* attrib_list)
80{
81 if (callback == NULL) {
82 _fegl_debugProc = NULL;
83 _fegl_debugFlags = fegl_DEBUG_CRITICAL_BIT | fegl_DEBUG_ERROR_BIT;
84 return EGL_SUCCESS;
85 }
86
87 if (attrib_list == NULL || attrib_list[0] == EGL_NONE) {
88 _fegl_debugProc = callback;
89 return EGL_SUCCESS;
90 }
91
92 EGLint debugFlags = _fegl_debugFlags;
93
94 for (int i = 0; attrib_list[i] != EGL_NONE; i += 2) {
95 EGLAttrib messageType = attrib_list[i];
96 EGLAttrib messageEnabled = attrib_list[i + 1];
97
98 EGLint mask = fegl_GetDebugTypeMask(messageType);
99 if (mask == 0 || (messageEnabled != EGL_TRUE && messageEnabled != EGL_FALSE))
100 return EGL_BAD_ATTRIBUTE;
101
102 debugFlags = (messageEnabled ? (debugFlags | mask) : (debugFlags & ~mask));
103 }
104
105 _fegl_debugProc = callback;
106 _fegl_debugFlags = debugFlags;
107 return EGL_SUCCESS;
108}

110/// Query an attribute from the debug layer.
111/// @param attribute Which attribute to query the value. Possible values are:
112/// - EGL_DEBUG_CALLBACK_KHR
113/// - EGL_DEBUG_MSG_CRITICAL_KHR
114/// - EGL_DEBUG_MSG_ERROR_KHR
115/// - EGL_DEBUG_MSG_WARN_KHR
116/// - EGL_DEBUG_MSG_INFO_KHR
117/// @param value Pointer to location where the result is written on success:
118/// - The current function callback if @a attribute is EGL_DEBUG_CALLBACK_KHR
119/// - Either EGL_TRUE or EGL_FALSE when @a attribute is a message type
120/// @returns EGL_TRUE on success, and EGL_FALSE on failure.
121///
122/// **Errors**
123/// - EGL_BAD_ATTRIBUTE if @a attribute is not one of the possible values.
124///
125EGLBoolean eglQueryDebugKHR(EGLint attribute, EGLAttrib* value)
126{
127 if (attribute == EGL_DEBUG_CALLBACK_KHR) {
128 *value = (EGLAttrib)_fegl_debugProc;
129 return EGL_TRUE;
130 }
131 EGLint mask = fegl_GetDebugTypeMask(attribute);
132 if (mask == 0) {
133 fegl_Error(EGL_BAD_ATTRIBUTE, __func__, NULL, "Unknown attribute");
134 return EGL_FALSE;
135 }
136 *value = (bool)(_fegl_debugFlags & mask);
137 return EGL_TRUE;
138}

159/// Returns the status (success or failure) of **the last EGL function call** in the current
160/// thread.
161/// @note Because eglGetError() is itself an EGL function, when calling the function twice, the
162/// second call will always return EGL_SUCCESS, indicating the status of the first call.
163/// @retval EGL_SUCCESS
164/// Function succeeded.
165/// @retval EGL_NOT_INITIALIZED
166/// EGL is not initialized, or could not be initialized, for the specified display.
167/// Any command may generate this error.
168/// @retval EGL_BAD_ACCESS
169/// EGL cannot access a requested resource (for example, a context is bound in another thread).
170/// Any command accessing a named resource may generate this error.
171/// @retval EGL_BAD_ALLOC
172/// EGL failed to allocate resources for the requested operation.
173/// Any command allocating resources may generate this error.
174/// @retval EGL_BAD_ATTRIBUTE
175/// An unrecognized attribute or attribute value was passed in an attribute list.
176/// Any command taking an attribute parameter or attribute list may generate this error.
177/// @retval EGL_BAD_CONTEXT
178/// An EGLContext argument does not name a valid EGLContext.
179/// Any command taking an EGLContext parameter may generate this error.
180/// @retval EGL_BAD_CONFIG
181/// An EGLConfig argument does not name a valid EGLConfig.
182/// Any command taking an EGLConfig parameter may generate this error.
183/// @retval EGL_BAD_CURRENT_SURFACE
184/// The current surface of the calling thread is a window, pbuffer or pixmap
185/// that is no longer valid.
186/// @retval EGL_BAD_DISPLAY
187/// An EGLDisplay argument does not name a valid EGLDisplay.
188/// Any command taking an EGLDisplay parameter may generate this error.
189/// @retval EGL_BAD_SURFACE
190/// An EGLSurface argument does not name a valid surface (window, pbuffer or pixmap)
191/// configured for rendering. Any command taking an EGLSurface parameter may generate this
192/// error.
193/// @retval EGL_BAD_MATCH
194/// Arguments are inconsistent; for example, an otherwise valid context requires buffers
195/// (e.g. depth or stencil) not allocated by an otherwise valid surface.
196/// @retval EGL_BAD_PARAMETER
197/// One or more argument values are invalid. Any command taking parameters may generate this
198/// error.
199/// @retval EGL_BAD_NATIVE_PIXMAP
200/// An EGLNativePixmapType argument does not refer to a valid native pixmap.
201/// Any command taking an EGLNativePixmapType parameter may generate this error.
202/// @retval EGL_BAD_NATIVE_WINDOW
203/// An EGLNativeWindowType argument does not refer to a valid native window.
204/// Any command taking an EGLNativeWindowType parameter may generate this error.
205/// @retval EGL_CONTEXT_LOST
206/// A power management event has occured. The application must destroy all contexts and
207/// reinitialise client API state and objects to continue rendering. Any command may generate
208/// this error.
209EGLint eglGetError()

211 EGLint lastStatus = _fegl_lastStatus;
212 return fegl_Success(), lastStatus;
213}
